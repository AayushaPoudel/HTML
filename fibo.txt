<script>
    const fibonacci = (n) => {
        let a = 0, b = 1, next;
        let result = "";

        for (let i = 1; i <= n; i++) {
            result += a + " ";
            next = a + b;
            a = b;
            b = next;
        }
        return result;
    };

    let terms = 10;
    document.write("Fibonacci Series: " + fibonacci(terms));
</script>
